use crate::{Signal16, Unit};

/// Built-in register chip acting as a keyboard
#[derive(Default)]
pub struct Keyboard {
    out: Signal16,
}

impl Keyboard {
    /// Returns the 16-bit character code of the currently pressed key on the
    /// physical board, or `0` if no key is pressed
    pub fn out(&self) -> Signal16 {
        self.out
    }
}

impl Keyboard {
    pub fn set(&mut self, keycode: sdl2::keyboard::Keycode) {
        self.out = to_ascii(keycode).into();
    }

    pub fn unset(&mut self, keycode: sdl2::keyboard::Keycode) {
        if self.out == to_ascii(keycode) {
            self.out = Signal16::FALSE;
        }
    }
}

impl Unit for Keyboard {
    fn eval(&mut self) {}
}

fn to_ascii(keycode: sdl2::keyboard::Keycode) -> i16 {

        Keycode::Slash => 47,

        Keycode::Num5 => 53,
        Keycode::Num6 => 54,
        Keycode::Num7 => 55,

        Keycode::Num9 => 57,
        Keycode::Colon => 58,
        Keycode::Semicolon => 59,
        Keycode::Less => 60,
        Keycode::Equals => 61,
        Keycode::Greater => 62,
        Keycode::Question => 63,
        Keycode::At => 64,

}
